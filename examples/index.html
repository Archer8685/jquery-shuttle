$("#resetBtn").bind("click", function() {
  $("#shuttleSel2").shuttle("reset");
});

$("#removeBtn").bind("click", function() {
  $("#shuttleSel2").shuttle("removeAll");
});

$("#disableBtn").bind("click", function() {
  var text = $(this).text();
  if (text === "Disable") {
    $("#shuttleSel2").shuttle("disable");
    $(this).html("Enable");
  } else if (text === "Enable") {
    $("#shuttleSel2").shuttle("enable");
    $(this).html("Disable");
  }
});
